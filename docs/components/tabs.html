Tabs are used to swap out entire sections of a page and clearly delineate which section you’re viewing with the .is-selected class.

{% highlight javascript linenos %} $(function() { $(".js-tabs-item").on("click", function(e) { var tabID = $(this).attr("aria-controls"); e.preventDefault(); e.stopPropagation(); $(this).siblings().removeClass("is-selected").attr({"aria-selected":"false"}); $(this).addClass("is-selected").attr({"tabindex":"0", "aria-selected":"true"}); $(".js-tabpanel").removeClass("d-block").addClass("d-none").attr({"aria-hidden":"true"}); $('[aria-labelledby="' + tabID + '"]').removeClass("d-none").addClass("d-block").attr({"aria-hidden":"false"}); }); }); {% endhighlight %} {% highlight html linenos %}
